<#
.SYNOPSIS
    Syncs tags from Resource Groups to all child resources.
    Resource tags are fully replaced by their parent Resource Group's tags.

.DESCRIPTION
    This Azure Automation Runbook:
      - Iterates over a defined list of target subscription IDs
      - For each Resource Group, retrieves its current tags
      - Applies those tags (full replace) to every resource within the group
      - Resources end up with exactly the tags on their Resource Group — no more, no less

.NOTES
    Authentication : System-assigned or User-assigned Managed Identity
    Required Role  : Tag Contributor (or Contributor) on each subscription/resource group
    Recommended    : Schedule once per day
#>

param (
    # Optional: restrict to a single resource group across all target subscriptions.
    # Leave empty to process all resource groups in each target subscription.
    [string]$TargetResourceGroupName = "",

    # Set to $true to log every resource evaluated, $false for summary only
    [bool]$VerboseLogging = $false
)

#region --- Target Subscription List ---
# Replace these GUIDs with your 7 actual subscription IDs.
# Keeping this as a plain array in the script body avoids Azure Automation param-parsing issues.
$TargetSubscriptionIds = @(
    "xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx",   # Sub 1 display name
    "yyyyyyyy-yyyy-yyyy-yyyy-yyyyyyyyyyyy",   # Sub 2 display name
    "zzzzzzzz-zzzz-zzzz-zzzz-zzzzzzzzzzzz",  # Sub 3 display name
    "aaaaaaaa-aaaa-aaaa-aaaa-aaaaaaaaaaaa",   # Sub 4 display name
    "bbbbbbbb-bbbb-bbbb-bbbb-bbbbbbbbbbbb",   # Sub 5 display name
    "cccccccc-cccc-cccc-cccc-cccccccccccc",   # Sub 6 display name
    "dddddddd-dddd-dddd-dddd-dddddddddddd"    # Sub 7 display name
)
#endregion

#region --- Authentication ---

Write-Output "=== Tag Inheritance Runbook Started: $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') ==="
Write-Output "Runtime reached. PowerShell version: $($PSVersionTable.PSVersion)"

try {
    Write-Output "Authenticating with Managed Identity..."
    Connect-AzAccount -Identity -ErrorAction Stop | Out-Null
    Write-Output "Authentication successful."
} catch {
    Write-Error "Authentication failed: $_"
    throw
}

#endregion

#region --- Validate Subscription List ---

if ($TargetSubscriptionIds.Count -eq 0) {
    Write-Error "TargetSubscriptionIds is empty. Provide at least one subscription ID."
    throw "No target subscriptions defined."
}

Write-Output "Target subscriptions defined: $($TargetSubscriptionIds.Count)"

$subscriptions = [System.Collections.Generic.List[object]]::new()

foreach ($subId in $TargetSubscriptionIds) {
    try {
        $sub = Get-AzSubscription -SubscriptionId $subId -ErrorAction Stop
        if ($sub.State -ne "Enabled") {
            Write-Warning "Subscription $subId is not in Enabled state (state: $($sub.State)). Skipping."
            continue
        }
        $subscriptions.Add($sub)
        Write-Output "  Validated: $($sub.Name) [$subId]"
    } catch {
        Write-Warning "  Could not retrieve subscription $subId — verify the ID and that the Managed Identity has access: $_"
    }
}

if ($subscriptions.Count -eq 0) {
    Write-Error "No valid, accessible subscriptions found from the provided list. Aborting."
    throw "No accessible target subscriptions."
}

Write-Output "Proceeding with $($subscriptions.Count) subscription(s)."

#endregion

#region --- Counters ---

$totalResources   = 0
$updatedResources = 0
$skippedResources = 0
$erroredResources = 0

#endregion

#region --- Main Processing Loop ---

foreach ($subscription in $subscriptions) {

    Write-Output ""
    Write-Output "--- Processing Subscription: $($subscription.Name) [$($subscription.Id)] ---"

    try {
        Set-AzContext -SubscriptionId $subscription.Id -ErrorAction Stop | Out-Null
    } catch {
        Write-Warning "Could not set context for subscription $($subscription.Id): $_"
        continue
    }

    # Get resource groups (optionally filtered)
    try {
        if ($TargetResourceGroupName) {
            $resourceGroups = Get-AzResourceGroup -Name $TargetResourceGroupName -ErrorAction Stop
        } else {
            $resourceGroups = Get-AzResourceGroup -ErrorAction Stop
        }
    } catch {
        Write-Warning "Could not retrieve resource groups in subscription $($subscription.Id): $_"
        continue
    }

    Write-Output "Found $($resourceGroups.Count) resource group(s)."

    foreach ($rg in $resourceGroups) {

        $rgName = $rg.ResourceGroupName
        $rgTags = $rg.Tags   # May be $null if no tags are set

        # Normalise: treat $null as empty hashtable so resources get their tags cleared
        if ($null -eq $rgTags) {
            $rgTags = @{}
        }

        Write-Output ""
        Write-Output "  [RG] $rgName — $($rgTags.Count) tag(s): $(($rgTags.Keys | Sort-Object) -join ', ')"

        # Get all resources in this resource group
        try {
            $resources = Get-AzResource -ResourceGroupName $rgName -ErrorAction Stop
        } catch {
            Write-Warning "  Could not list resources in '$rgName': $_"
            continue
        }

        foreach ($resource in $resources) {

            $totalResources++

            # Current tags on the resource (normalise $null → empty hashtable)
            $currentTags = $resource.Tags
            if ($null -eq $currentTags) {
                $currentTags = @{}
            }

            # Check whether an update is needed (compare sorted key=value pairs)
            $currentSnapshot = ($currentTags.GetEnumerator() | Sort-Object Key | ForEach-Object { "$($_.Key)=$($_.Value)" }) -join "|"
            $desiredSnapshot  = ($rgTags.GetEnumerator()     | Sort-Object Key | ForEach-Object { "$($_.Key)=$($_.Value)" }) -join "|"

            if ($currentSnapshot -eq $desiredSnapshot) {
                if ($VerboseLogging) {
                    Write-Output "    [SKIP]    $($resource.ResourceId) — tags already match"
                }
                $skippedResources++
                continue
            }

            # Apply the resource group's tags, completely replacing whatever was there
            try {
                Update-AzTag `
                    -ResourceId $resource.ResourceId `
                    -Tag        $rgTags `
                    -Operation  Replace `
                    -ErrorAction Stop | Out-Null

                if ($VerboseLogging) {
                    Write-Output "    [UPDATED] $($resource.ResourceId)"
                }
                $updatedResources++

            } catch {
                Write-Warning "    [ERROR]   $($resource.ResourceId) — $_"
                $erroredResources++
            }
        }
    }
}

#endregion

#region --- Summary ---

Write-Output ""
Write-Output "=== Runbook Completed: $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') ==="
Write-Output "  Total resources evaluated : $totalResources"
Write-Output "  Updated                   : $updatedResources"
Write-Output "  Already in sync (skipped) : $skippedResources"
Write-Output "  Errors                    : $erroredResources"

if ($erroredResources -gt 0) {
    Write-Warning "Some resources failed to update. Review warnings above for details."
}

#endregion
