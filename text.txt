SELECT * FROM standing_orders WHERE ((frequency = 'Daily' AND (lastprocessed IS NULL OR lastprocessed < CURRENT_DATE - INTERVAL '1 day')) OR (frequency = 'Weekly' AND (lastprocessed IS NULL OR lastprocessed < CURRENT_DATE - INTERVAL '1 week')) OR (frequency = 'Monthly' AND (lastprocessed IS NULL OR lastprocessed < CURRENT_DATE - INTERVAL '1 month')))
