// Function to verify the PIN during a transaction
function verifyPIN(inputPIN, storedSalt, storedHashedPIN) {
    // Hash the input PIN with the stored salt
    const hashedInputPIN = hashPIN(inputPIN, storedSalt);

    // Compare the hashed input PIN with the stored hashed PIN
    return hashedInputPIN === storedHashedPIN;
}
